### Task 4: Fix game-log.js — character filter include-list logic

**Files:**
- Modify: `game/static/js/game-log.js`

**Context:**
Current bug: `filters.characters` is used as an exclude-list (IDs to hide) in
`updateCharacterFilter` but should be an include-list (IDs to show, empty = all).

New semantics:
- `filters.characters` = Set of character IDs to **show** (empty = show all)
- When a character is unchecked from "all checked" state: build the include set from
  all *still-checked* characters
- When a character is rechecked: add to set; if all now checked, clear the set
- When "All" is checked: clear set (show all)
- Events with no `character_id` (DM/system) always show regardless of filter

**Step 1: Replace `updateCharacterFilter` checkbox handler (lines ~245-271)**

Find:
```javascript
        dropdown.querySelectorAll('input[type="checkbox"]').forEach(function (checkbox) {
            checkbox.addEventListener("change", function (e) {
                if (e.target.value === "all") {
                    var isChecked = e.target.checked;
                    dropdown.querySelectorAll('input[type="checkbox"]').forEach(function (cb) {
                        cb.checked = isChecked;
                    });
                    self.filters.characters.clear();
                } else {
                    var charId = parseInt(e.target.value);

                    if (e.target.checked) {
                        self.filters.characters.delete(charId);
                    } else {
                        self.filters.characters.add(charId);
                    }

                    var allCheckbox = dropdown.querySelector('input[value="all"]');
                    var allChecked = dropdown.querySelectorAll('input:not([value="all"]):checked').length ===
                                     dropdown.querySelectorAll('input:not([value="all"])').length;
                    allCheckbox.checked = allChecked;
                }

                self.updateCharacterButtonText();
                self.applyFilters();
            });
        });
```

Replace with:
```javascript
        dropdown.querySelectorAll('input[type="checkbox"]').forEach(function (checkbox) {
            checkbox.addEventListener("change", function (e) {
                if (e.target.value === "all") {
                    var isChecked = e.target.checked;
                    dropdown.querySelectorAll('input[type="checkbox"]').forEach(function (cb) {
                        cb.checked = isChecked;
                    });
                    // Checked = show all; unchecked = also show all (no partial state from "All")
                    self.filters.characters.clear();
                } else {
                    var charId = parseInt(e.target.value);
                    var allIndividual = dropdown.querySelectorAll('input:not([value="all"])');
                    var checkedIndividual = dropdown.querySelectorAll('input:not([value="all"]):checked');

                    if (e.target.checked) {
                        // Re-checking: add to include set; if all now checked, clear (show all)
                        self.filters.characters.add(charId);
                        if (checkedIndividual.length === allIndividual.length) {
                            self.filters.characters.clear();
                        }
                    } else {
                        // Unchecking: if filter was empty (show all), build set from still-checked
                        if (self.filters.characters.size === 0) {
                            checkedIndividual.forEach(function (cb) {
                                self.filters.characters.add(parseInt(cb.value));
                            });
                        } else {
                            self.filters.characters.delete(charId);
                        }
                    }

                    var allCheckbox = dropdown.querySelector('input[value="all"]');
                    allCheckbox.checked = self.filters.characters.size === 0;
                }

                self.updateCharacterButtonText();
                self.applyFilters();
            });
        });
```
